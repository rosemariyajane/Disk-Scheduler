# ------------------------------------------------
# Utility Functions
# ------------------------------------------------

def calculateMetrics(sequence, seek_time):
    count = len(sequence) - 1
    avg_seek = seek_time / count if count > 0 else 0
    throughput = count / seek_time if seek_time > 0 else 0
    return avg_seek, throughput


def generateGraph(sequence):
    plt.figure(figsize=(8, 5))
    plt.plot(range(len(sequence)), sequence, marker='o', linestyle='-', linewidth=2)
    plt.xlabel("Seek Step")
    plt.ylabel("Cylinder Number")
    plt.title("Disk Seek Sequence")
    plt.grid(True)

    if not os.path.exists("static"):
        os.makedirs("static")

    plt.savefig("static/seek_sequence.png")
    plt.close()


# ------------------------------------------------
# Disk Scheduling Algorithms
# ------------------------------------------------

def fcfs(requests, head):
    sequence = [head] + requests
    seek = sum(abs(sequence[i] - sequence[i+1]) for i in range(len(sequence)-1))
    return sequence, seek


def sstf(requests, head):
    pending = requests.copy()
    sequence = [head]
    total_seek = 0

    while pending:
        nearest = min(pending, key=lambda x: abs(x - head))
        total_seek += abs(head - nearest)
        head = nearest
        sequence.append(head)
        pending.remove(nearest)

    return sequence, total_seek


def scan(requests, head, end=199):
    left = [r for r in requests if r < head]
    right = [r for r in requests if r >= head]

    left.sort()
    right.sort()

    sequence = [head] + right + [end] + left[::-1]
    seek = sum(abs(sequence[i] - sequence[i+1]) for i in range(len(sequence)-1))
    return sequence, seek


def cScan(requests, head, end=199):
    left = [r for r in requests if r < head]
    right = [r for r in requests if r >= head]

    left.sort()
    right.sort()

    # Go right → jump to 0 → go left
    sequence = [head] + right + [end, 0] + left

    seek = sum(abs(sequence[i] - sequence[i+1]) for i in range(len(sequence)-1))
    return sequence, seek


# ------------------------------------------------
# Flask Routes
# ------------------------------------------------

@app.route('/')
def index():
    return HTML_PAGE


@app.route('/schedule', methods=['POST'])
def schedule():
    data = request.json
    requests = list(map(int, data["requests"].split(",")))
    head = int(data["head"])
    algo = data["algorithm"]

    if algo == "FCFS":
        seq, seek = fcfs(requests, head)
    elif algo == "SSTF":
        seq, seek = sstf(requests, head)
    elif algo == "SCAN":
        seq, seek = scan(requests, head)
    elif algo == "C-SCAN":
        seq, seek = cScan(requests, head)
    else:
        return jsonify({"error": "Invalid algorithm selected"}), 400

    avg, tput = calculateMetrics(seq, seek)
    generateGraph(seq)

    return jsonify({
        "seek_sequence": seq,
        "seek_time": seek,
        "avg_seek_time": avg,
        "throughput": tput
    })
